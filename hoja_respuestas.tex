%───────────────────────────────────────────────────────────────────────────────
% GEOMETRIA DE LA GRILLA (todo en pt, 1mm = 2.8346pt)
%
% Ancho hoja carta: 216mm = 612pt
% Margenes: izq 1.2cm + der 1.1cm = 2.3cm = 65.2pt
% Area de texto: 612 - 65.2 = 546.8pt
%
% columnsep = 5pt  →  4 separadores = 20pt
% Ancho por columna = (546.8 - 20) / 5 = 105.4pt
%
% Cada fila:  TM  +  gap  +  Nwidth  +  gap  +  4×Bdia  +  3×Bsep
%              6  +   1   +    14    +   2   +  4×19.0  +  3×2
%            = 6 + 1 + 14 + 2 + 76 + 6 = 105pt  ← encaja exacto
%───────────────────────────────────────────────────────────────────────────────
\setlength{\columnsep}{5pt}
\setlength{\columnseprule}{0.3pt}

% Medidas de la fila
\newlength{\TMw}    \setlength{\TMw}{6pt}    % ancho del timing mark
\newlength{\Nw}     \setlength{\Nw}{14pt}    % ancho fijo del numero (1-3 digitos)
\newlength{\Bdia}   \setlength{\Bdia}{19pt}  % celda de cada burbuja (ancho = alto)
\newlength{\Bsep}   \setlength{\Bsep}{2pt}   % separacion entre celdas de burbuja
\newlength{\Brad}   \setlength{\Brad}{9pt}   % radio dibujado (< Bdia/2 = 9.5)
\newlength{\Rowsep} \setlength{\Rowsep}{0pt} % espacio extra entre filas

% Offset horizontal acumulado desde el inicio de la linea hasta el
% borde izquierdo de la primera burbuja:
%   TMw(6) + gap(1) + Nw(14) + gap(2) = 23pt
% El CENTRO de la primera burbuja esta en: 23 + Bdia/2 = 23 + 9.5 = 32.5pt

%───────────────────────────────────────────────────────────────────────────────
% MACRO: burbuja OMR
%───────────────────────────────────────────────────────────────────────────────
\newcommand{\B}{%
  \makebox[\Bdia][c]{%
    \tikz[baseline=-3.5pt]{%
      \filldraw[fill=black!10, draw=black!60, line width=0.6pt]
        (0,0) circle (\Brad);}%
  }%
}

%───────────────────────────────────────────────────────────────────────────────
% MACRO: timing mark (rectangulo negro solido)
%───────────────────────────────────────────────────────────────────────────────
\newcommand{\TM}{%
  \makebox[\TMw][c]{%
    \tikz[baseline=-4pt]{\fill[black](0,-5pt) rectangle (\TMw,5pt);}%
  }%
}

%───────────────────────────────────────────────────────────────────────────────
% MACRO: fila de pregunta
% Estructura: [TM][1pt][  Nw  ][2pt][B][2pt][B][2pt][B][2pt][B]
% El numero usa \makebox[\Nw][r] → siempre alineado a la derecha dentro
% de un cuadro de ancho fijo.  Las burbujas quedan en X fija para 1, 2 o 3 digitos.
%───────────────────────────────────────────────────────────────────────────────
\newcommand{\Q}[1]{%
  \noindent
  \TM
  \hspace{1pt}%
  \makebox[\Nw][r]{\scriptsize\textbf{#1.}}%
  \hspace{2pt}%
  \B\hspace{\Bsep}\B\hspace{\Bsep}\B\hspace{\Bsep}\B%
  \par\vspace{\Rowsep}%
}

%───────────────────────────────────────────────────────────────────────────────
% MACRO: encabezado A B C D
% El indent = TMw + 1 + Nw + 2 = 23pt coloca la primera letra exactamente
% encima del centro de la primera burbuja (23 + Bdia/2 = 32.5pt en ambos casos).
%───────────────────────────────────────────────────────────────────────────────
\newcommand{\HDR}{%
  \noindent
  \hspace{\TMw}\hspace{1pt}\hspace{\Nw}\hspace{2pt}%
  \makebox[\Bdia][c]{\footnotesize\textbf{A}}%
  \hspace{\Bsep}%
  \makebox[\Bdia][c]{\footnotesize\textbf{B}}%
  \hspace{\Bsep}%
  \makebox[\Bdia][c]{\footnotesize\textbf{C}}%
  \hspace{\Bsep}%
  \makebox[\Bdia][c]{\footnotesize\textbf{D}}%
  \par\vspace{2pt}%
}

%───────────────────────────────────────────────────────────────────────────────
% MACRO: burbuja pequena para grilla de ID
%───────────────────────────────────────────────────────────────────────────────
\newcommand{\Bs}{%
  \tikz[baseline=-2pt]{%
    \filldraw[fill=black!10, draw=black!55, line width=0.4pt]
      (0,0) circle (3.5pt);}%
}
